
A)Basic:
docker run -d --name db -e POSTGRES_PASSWORD=password postgres

docker run -d --name pgadmin -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=password dpage/pgadmin4

docker exec -it pgadmin ping db

B)With network:
1)network create my-network

2)docker run -d --name db --network my-network -e POSTGRES_PASSWORD=password postgres

3)docker run -d --name pgadmin --network my-network -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=password dpage/pgadmin4

C)FULL Commands:

1)Start the Posgres service
docker run -d \
  --name postgres_container \
  -e POSTGRES_USER=admin \
  -e POSTGRES_PASSWORD=admin \
  -e PGDATA=/data/postgres \
  -v postgres:/data/postgres \
  -p 5432:5432 \
  --network postgres \
  --restart unless-stopped \
  postgres

2)Start PGAdmin service
docker run -d \
  --name pgadmin_container \
  -e PGADMIN_DEFAULT_EMAIL=pgadmin@pgadmin.org \
  -e PGADMIN_DEFAULT_PASSWORD=admin \
  -e PGADMIN_CONFIG_SERVER_MODE = False \
  -v pgadmin:/var/lib/pgadmin \
  -p 5050:80 \
  --network postgres \
  --restart unless-stopped \
  dpage/pgadmin4
